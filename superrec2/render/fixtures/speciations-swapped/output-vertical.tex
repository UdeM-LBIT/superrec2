        draw={#1}, fill={species background color!50!white},
        align=center,

        line width={0.5pt},

    branch node/.default={black},
    speciation/.style={
        branch node={#1}, rectangle, rounded corners,

        minimum width={8},
        minimum height={8},

        minimum width={8},
        minimum height={8},

        chamfered rectangle sep={8 / 2.4},
        inner xsep=2pt,
        inner ysep=-1pt,
        minimum width={8},
        minimum height={8},
    },
